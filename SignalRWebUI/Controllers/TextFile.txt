        function showproductid(value) {
            var id = value;
            $.ajax({
                type: 'post',
                data: { item:id },
                contentType: 'json',
                success: function () {
                    Swal.fire({
                        title: "Tebrikler",
                        text: "Tıkladığınız Ürün ID Değeri: " + id,
                        icon: "success"
                    });
                }
            });
        }
